@startuml
autonumber
actor "Administrator" as ADMIN

activate ADMIN
ADMIN -> ":Plataform" : access Platform
activate ":Plataform"
":Plataform" --> ADMIN : asks the admin if he wants to Create, Read, Edit or Delete 
deactivate ":Plataform"

ADMIN -> ":Plataform" : choses Create
activate ":Plataform"
":Plataform" --> ADMIN : asks for Parking Space attributes \n(designacaoUnica, andar, preco, parqueEstacionamento)\n 
deactivate ":Plataform"

ADMIN -> ":Plataform" : inserts the asked info
activate ":Plataform"
":Plataform" --> ADMIN : shows the info and asks for confirmation
deactivate ":Plataform"

ADMIN -> ":Plataform" : confirms
activate ":Plataform"
":Plataform" --> ADMIN : successful operation
deactivate ":Plataform"

deactivate ADMIN
@enduml
